#function from https://community.rstudio.com/t/convert-tibble-into-matrix/68262/2
make_matrix <- function(df,rownames = NULL){
  my_matrix <-  as.matrix(df)
  if(!is.null(rownames))
    rownames(my_matrix) = rownames
  my_matrix
}
